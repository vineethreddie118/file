generateTimeDropdown() {
  const start = new Date();
  start.setHours(1, 0, 0); // 01:00 AM

  const end = new Date();
  end.setHours(23, 30, 0); // 11:30 PM

  while (start <= end) {
    const timeString = start.toTimeString().substring(0, 5); // Get time in HH:mm format
    this.availableTimes.push(timeString);
    start.setMinutes(start.getMinutes() + 30); // Increment by 30 minutes
  }
}

<div class="form-group col-md-2">
  <select class="form-control form-control-custom" id="openTime{{i}}" name="openTime{{i}}"
    [(ngModel)]="oCurrentProviderLocationInfo.providerSvcLocHours[i].openTimeValue">
    <option *ngFor="let time of availableTimes" [value]="time">{{ time }}</option>
  </select>
</div>
