 private fnFetchData(): void {
    this.bIsLoading = true;

    let arrCallsToMake = [
      this.searchService.getProviderLocationAlternateIdLookup().pipe(
        catchError((err) => {
          this.bIsLoading = false;
          this.addService.fnRaiseGenericAlert(
            'getProviderLocationAlternateIdLookup'
          );
          return of(null);
        })
      ),
      this.searchService
        .getSoloProviderLocationAlternateId(this.nProviderId)
        .pipe(
          catchError((err) => {
            this.bIsLoading = false;
            this.addService.fnRaiseGenericAlert(
              'getSoloProviderLocationAlternateId'
            );
            return of(null);
          })
        ),
      this.searchService.getPracticeLocationDetails(this.nProviderId).pipe(
        catchError((err) => {
          this.bIsLoading = false;
          this.addService.fnRaiseGenericAlert('getPracticeLocationDetails');
          return of(null);
        })
      ),
    ];

    forkJoin([...arrCallsToMake])
      .pipe(
        map(([arrLookup, oAlternateIdInfo, oPracticeLocationInfo]) => {
          if (!!arrLookup) {
            this.arrLookup = { ...arrLookup[0] };
          }
          if (!!oAlternateIdInfo) {
            this.oAlternateIdsInfo = { ...oAlternateIdInfo };
          }
          if (!!oPracticeLocationInfo) {
            this.oProviderLocation = { ...oPracticeLocationInfo };
          }
         

          // Linking the billing locations to alternate Id
          this.oAlternateIdsInfo.providerLocationAlternateIds.forEach(altId=>{
            this.linkBillingLocsToPracticeLocs(altId);
          })

          this.oProviderLocation.providerLocationInfo?.forEach(
            (location) => {
              location.bIsChecked = false;
              location.isClickable = false;
            }
          );
          
          this.bIsLoading = false;
        })
      )
      .subscribe();
  }

 onSubmitAlernateIdForm(dupDialog: any): void {
    this.formTemplate.triedToSubmit = true;
    if (this.formData.invalid || this.formTemplate.data.selectedParticipationLocation.length===0) {
      return;
    }

    const data = this.formTemplate.data;
    let flagAltLocAlreadyExists = false;
    let flagInvalidIdNumber = false;

    const oValidStatus = this.fnCheckForValidValue(data.idType.locationIdTypeId, data.locationAltId);
    if(false === oValidStatus.bIsValid) {
      this.confirmationDialogService
        .alert({ title: 'Invalid Value',
                   message: oValidStatus.sMsg
                  }).then((bIsConfirmed) => {});
        return;

    }

    data.selectedParticipationLocation.forEach((participationLocation) => {
      const newAltLoc = {
        locationAlternateXrefId: this.formTemplate.editMode
          ? this.oAlternateIdsInfo.providerLocationAlternateIds[
              this.formTemplate.indexOfEdited
            ].locationAlternateXrefId
          : 0,
        providerServiceLocationMapId:
          participationLocation.providerServiceLocationMapId,
        serviceLocationId: participationLocation.serviceLocationId,
        locationIdTypeId: data.idType.locationIdTypeId,
        locationAltId: data.locationAltId,
        locationIdTypeDesc: data.idType.locationIdType,
        effectiveDate: data.effectiveDate + 'T00:00:00',
        expirationDate: data.expirationDate + 'T00:00:00',
        userId: participationLocation.userId,
        state: data.state,
        locationName: participationLocation.locationName,
        address1: participationLocation.serviceAddress,
        address2: participationLocation.address2,
        city: participationLocation.city,
        county: participationLocation.county,
        stateCode: participationLocation.stateCode,
        zip: participationLocation.zip,
      };

      const locationAltIdArr = newAltLoc.locationAltId.split('');

      if(newAltLoc.locationIdTypeDesc==='MEDICAID ID' 
        && locationAltIdArr.length > 1
        && new Set(locationAltIdArr).size===1 
        && !isNaN(parseFloat(locationAltIdArr[0]))
        ){
        flagInvalidIdNumber = true;
        return;
      }
  
      this.oAlternateIdsInfo.providerLocationAlternateIds?.forEach(
        (locAlt, i) => {
          if (
            locAlt.locationIdTypeId === newAltLoc.locationIdTypeId &&
            locAlt.locationIdTypeDesc === newAltLoc.locationIdTypeDesc &&
            locAlt.locationName === newAltLoc.locationName &&
            locAlt.locationAltId === newAltLoc.locationAltId &&
            locAlt.serviceLocationId === newAltLoc.serviceLocationId &&
            !(this.formTemplate.editMode && i === this.formTemplate.indexOfEdited)
          ) {
            flagAltLocAlreadyExists = true;
            return;
          }
        }
      );
    });

    if (flagInvalidIdNumber) {
      this.confirmationDialogService
        .confirm({ title: 'Invalid Id Number', 
                  message: 'ID Number for Id type, "MEDICAID ID" cannot be a single digit repeated'}
                  ).then((bIsConfirmed) => {
          
        });
        return;
    }

    if (flagAltLocAlreadyExists) {
      //open modal to display that an alternate id already exists with same details
      this.oBsModalSvc.open(dupDialog);
      return;
    }

    data.selectedParticipationLocation.forEach((participationLocation) => {
      const newAltLoc = {
        locationAlternateXrefId: this.formTemplate.editMode
          ? this.oAlternateIdsInfo.providerLocationAlternateIds[
              this.formTemplate.indexOfEdited
            ].locationAlternateXrefId
          : 0,
        providerServiceLocationMapId:
          participationLocation.providerServiceLocationMapId,
        serviceLocationId: participationLocation.serviceLocationId,
        locationIdTypeId: data.idType.locationIdTypeId,
        locationAltId: data.locationAltId,
        locationIdTypeDesc: data.idType.locationIdType,
        effectiveDate: data.effectiveDate + 'T00:00:00',
        expirationDate: data.expirationDate? data.expirationDate+ 'T00:00:00': null,
        userId: participationLocation.userId,
        state: data.state,
        locationName: participationLocation.locationName,
        address1: participationLocation.serviceAddress,
        address2: participationLocation.address2,
        city: participationLocation.city,
        county: participationLocation.county,
        stateCode: participationLocation.stateCode,
        zip: participationLocation.zip,
      };

      //linking billing locations to provider location
      this.linkBillingLocsToPracticeLocs(newAltLoc);

      if (this.formTemplate.editMode) {
        let index = this.formTemplate.indexOfEdited;

        this.oAlternateIdsInfo.providerLocationAlternateIds[index] = newAltLoc;
        // this.arrLocationInfo[index] = participationLocation;

        this.formTemplate.editMode = false;
      } else {
        if (this.oAlternateIdsInfo.providerLocationAlternateIds) {
          this.oAlternateIdsInfo.providerLocationAlternateIds.push(newAltLoc);
        } else {
          this.oAlternateIdsInfo.providerLocationAlternateIds = [newAltLoc];
        }
        // this.arrLocationInfo.push(participationLocation);
      }
    });

    this.formTemplate.showForm = false;
    this.dataChanged = true;
    this.formTemplate.triedToSubmit = false;
    this.formTemplate.indexOfEdited = null;

    this.formData.reset();
    this.fnResetAltLocationSelect();

  }

private linkBillingLocsToPracticeLocs(pracLoc: any){
    for(let provLoc of this.oProviderLocation.providerLocationInfo){
      if(provLoc.serviceLocationId===pracLoc.serviceLocationId){
        pracLoc.providerLocBillingInfo = [...provLoc.providerLocBillingInfo];
        break;
      }
    }
  }
