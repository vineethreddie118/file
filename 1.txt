 public fnHandleAddressChange(e: any): void {
    if (e.item) {
    this.providerForm.address1 = e.item.street_line.length>25 ? e.item.street_line.trim().substring(0,25): e.item.street_line;
    this.providerForm.address2 = e.item.street_line.length>25 ? (e.item.street_line.trim().substring(25) + `${e.item.secondary? ', '+e.item.secondary: e.item.secondary} `): e.item.secondary;
    this.providerForm.city = e.item.city;
    this.providerForm.stateCode = e.item.state;
    this.providerForm.zip = e.item.zipcode;
  } else {
    this.providerForm.address1 = e.target.value;
  }
    this.bIsLocationValidated = false;
    this.oAddressChanged = {...this.oAddressChanged, val: true};
    this.fnSetAddress();
  }

private fnFetchData(): void {
    let arrCallsToMake = [
      null === this.providerLookupDetails ? 
        this.searchService.getProviderProfileLookup().pipe(catchError((err) => {
          this.bIsLoading = false;
          this.addService.fnRaiseGenericAlert('getLookupDetails');
          return of(null);
        })) : 
        of(null),
      this.providerForm.providerId !== undefined ? 
        this.searchService.getProfileInformation(this.providerForm.providerId).pipe(catchError((err) => {
          this.bIsLoading = false;
          this.addService.fnRaiseGenericAlert('getProfileInformation');
          return of(null)
        })) : 
        of(null)
    ];

    this.bIsLoading = true;
    forkJoin([...arrCallsToMake]).pipe(
      map(([lookupInfo, providerInfo]) => {
        if(!!lookupInfo) {
          this.providerLookupDetails = {...lookupInfo[0]};
        }
        
        if (!!providerInfo) {
          this.providerForm = { ...providerInfo };
          this.providerForm.providerId = this.nProviderId;
          this.bIsLocationValidated = this.providerForm['addressValidated'];
          this.savetoFlexcare = this.providerForm['savetoFlexcare'];
         
          if (!!this.providerForm.providerAlternateIDSection && this.providerForm.providerAlternateIDSection.length > 0) {
            this.providerForm.providerAlternateIDSection[0].medicare = new Medicare();
            if(this.providerForm.providerAlternateIDSection[0].providerNpi[0].providerTaxonomy && this.providerForm.providerAlternateIDSection[0].providerNpi[0].providerTaxonomy.length === 0){
              this.providerForm.providerAlternateIDSection[0].providerNpi[0].providerTaxonomy= []
            }else{
              this.providerForm.providerAlternateIDSection[0].providerNpi[0].providerTaxonomy.forEach((taxonamy, i) =>{
                if (taxonamy.taxonomyCode.length) {
                  this.disabledTaxonomies.push(taxonamy.taxonomyCode);
                }
              } )
            }
          }
          this.defaultProviderForm = JSON.parse(JSON.stringify(this.providerForm));
          this.fnSetAddress();
        }
        this.bIsLoading = false;
      })
    ).subscribe();
  }
