 public fnHandleAddressChange(e: any): void {
    if (e.item) {
    this.providerForm.address1 = e.item.street_line;
    this.providerForm.address2 = e.item.secondary;
    this.providerForm.city = e.item.city;
    this.providerForm.stateCode = e.item.state;
    this.providerForm.zip = e.item.zipcode;
  } else {
    this.providerForm.address1 = e.target.value;
  }
    this.bIsLocationValidated = false;
    this.oAddressChanged = {...this.oAddressChanged, val: true};
  }

  public fnGetSmartyStInfo(): SmartyStreetAddressInfo {
    const oSmartyStInfo: SmartyStreetAddressInfo = new SmartyStreetAddressInfo();
    oSmartyStInfo.street = this.providerForm.address1;
    oSmartyStInfo.street2 = this.providerForm.address2;
    oSmartyStInfo.city =  this.providerForm.city;
    oSmartyStInfo.state =  this.providerForm.stateCode;
    oSmartyStInfo.zipCode =  this.providerForm.zip;

    return oSmartyStInfo;
  }

  public fnProfileHandleSmartyStAddressUpdatedEvent(oUpdatedAddressInfo: SmartyStreetUpdatedInfo) {
    this.bIsMissingSecondaryInfo = oUpdatedAddressInfo.bIsMissingSecondaryInfo;
    this.bIsLocationValidated = oUpdatedAddressInfo.bIsServiceAddressValidated;

    if (!!oUpdatedAddressInfo.street)
      this.providerForm.address1 = oUpdatedAddressInfo.street;

    if (!!oUpdatedAddressInfo.city)
      this.providerForm.city = oUpdatedAddressInfo.city;

    if (!!oUpdatedAddressInfo.zip)
      this.providerForm.zip = oUpdatedAddressInfo.zip;

    if (!!oUpdatedAddressInfo.county)
      this.providerForm.county = oUpdatedAddressInfo.county;
  }

  public fnHandleInvalidAddressUserChoice(choice: any): void {
    if(1 === choice) {
      this.bIsInvalidAddressAllowed = true;
    }
  }
