fnAddLocationDetails() {
    
    
    if ( !this.bIsServiceAddressValidated && !this.bIsInvalidAddressAllowed){
      this.addService.fnRaiseWarningDlg("Invalid" ," Please enter valid address");
      return;
    }
    if (!this.oCurrentProviderLocationInfo.phone && this.oCurrentProviderLocationInfo.phoneExt) {
      this.addService.fnRaiseWarningDlg("Missing Phone", "Please enter the phone number.");
      return;
  }
    let timeValueInvalid = false;
    const hasImproperNextDayHours = this.oCurrentProviderLocationInfo.providerSvcLocHours.some(hour => {
      if ((hour.openTimeValue   && hour.closeTimeValue === "") || (hour.openTimeValue === '' && hour.closeTimeValue) ) {
        //if start and end date as null skip it because they are default values
        timeValueInvalid = true;
      }
   
      const openHour = parseInt(hour.openTimeValue.split(":")[0], 10); // Convert open time hour part to integer
      const closeHour = parseInt(hour.closeTimeValue.split(":")[0], 10); // Convert close time hour part to integer
  
      return openHour > closeHour || (openHour === closeHour && hour.openTimeValue >= hour.closeTimeValue); // Checks if time spans over to next day improperly
    });
    if(timeValueInvalid){
      this.addService.fnRaiseWarningDlg("Incomplete Hours" ," Please Fill both open and close time");
      return;
    }
  
    if (hasImproperNextDayHours) {
      this.addService.fnRaiseWarningDlg("Invalid Hours", "End time has been exceed for the day.");
      return; // Stop it if invalid hours are shown
    }
    const hasOverlappingHours = this.oCurrentProviderLocationInfo.providerSvcLocHours.some((currentHour, index, array) => {
      const currentOpen = currentHour.openTimeValue;
      const currentClose = currentHour.closeTimeValue;
      return array.some((checkHour, checkIndex) => {
          if (index === checkIndex) return false; // Don't compare with itself
          if (currentHour.dayOfWeek !== checkHour.dayOfWeek) return false; // Only compare same days

          const checkOpen = checkHour.openTimeValue;
          const checkClose = checkHour.closeTimeValue;
          // Check for overlap
          return (currentOpen < checkClose && currentClose > checkOpen); // Overlap condition
      });
  });

  if (hasOverlappingHours) {
      
      this.addService.fnRaiseWarningDlg("Overlap Detected", "Office hours cannot overlap for the same day.");
    
      
      return; // Stop it  if overlap detected
  }
